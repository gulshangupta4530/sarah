Muli-Dimensional Array Questions

1. Enter the four number from user and print it into matrix

Code:

#include <stdio.h>
void main() {
    int mat[2][2], i, j;
    printf("Enter 4 numbers: ");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            scanf("%d", &mat[i][j]);
        }
    }
    printf("Matrix:\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            printf("%d\t", mat[i][j]);
        }
        printf("\n");
    }
}

Output:

Enter 4 numbers: 1 2 3 4
Matrix:
1       2
3       4

2.Enter the matrix from and transpose them ?

Code:

#include <stdio.h>
void main() {
    int mat[2][2], trans[2][2], i, j;
    printf("Enter 4 numbers for 2x2 matrix: ");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            scanf("%d",&mat[i][j]);
            trans[j][i] = mat[i][j];
        }
    }
    printf("Transpose Matrix:\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            printf("%d\t", trans[i][j]);
        }
        printf("\n");
    }
}

Output:

Enter 4 numbers for 2x2 matrix: 1 2 3 4      
Transpose Matrix:
1       3
2       4

5.Enter 2 matrix from user and multiply them ?

Code:

#include <stdio.h>
void main() {
    int a[2][2], b[2][2], mul[2][2], i, j, k;
    printf("Enter 4 numbers for first matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&a[i][j]);
    printf("Enter 4 numbers for second matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&b[i][j]);
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            mul[i][j]=0;
            for(k=0;k<2;k++)
                mul[i][j] += a[i][k]*b[k][j];
        }
    }
    printf("Multiplication Matrix:\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++)
            printf("%d\t", mul[i][j]);
        printf("\n");
    }
}

Output:

Enter 4 numbers for first matrix: 1 2 3 4
Enter 4 numbers for second matrix: 1 2 3 4
Multiplication Matrix:
7       10
15      22
